        // ========== 本地音频核心控制：播放/暂停100%稳定 ==========
        const audio = document.getElementById('localMusic');
        const musicBtn = document.getElementById('musicBtn');
        let isPlaying = false;

        // 监听音频原生事件，绝对同步状态（彻底解决暂停失效）
        audio.addEventListener('play', () => {
            isPlaying = true;
            musicBtn.innerText = "⏸";
        });
        audio.addEventListener('pause', () => {
            isPlaying = false;
            musicBtn.innerText = "▶";
        });
        // 监听音频结束，自动循环（兜底）
        audio.addEventListener('ended', () => {
            audio.play();
        });

        // 播放/暂停切换按钮
        musicBtn.addEventListener('click', () => {
            if (isPlaying) {
                audio.pause();
            } else {
                // 激活音频上下文，解决浏览器所有播放限制
                const ac = new (window.AudioContext || window.webkitAudioContext)();
                ac.resume();
                audio.play();
            }
        });

        // ========== 原有点击音效：保留，与背景音乐无冲突 ==========
        let clickAudioContext, clickSoundBuffer;
        const CLICK_SOUND_URL = 'https://assets.mixkit.co/sfx/preview/mixkit-soft-click-1113.mp3';

        // 初始化点击音效
        function initClickSound() {
            if (!clickAudioContext) {
                clickAudioContext = new (window.AudioContext || window.webkitAudioContext)();
                fetch(CLICK_SOUND_URL)
                    .then(res => res.arrayBuffer())
                    .then(buf => clickAudioContext.decodeAudioData(buf))
                    .then(data => clickSoundBuffer = data)
                    .catch(err => console.log("点击音效加载失败：", err));
            }
        }

        // 播放点击音效
        function playClickSound() {
            if (!clickAudioContext || !clickSoundBuffer) return;
            const source = clickAudioContext.createBufferSource();
            source.buffer = clickSoundBuffer;
            const gainNode = clickAudioContext.createGain();
            gainNode.gain.value = 0.4; // 音量适中，不突兀
            source.connect(gainNode);
            gainNode.connect(clickAudioContext.destination);
            source.start(0);
        }

        // 页面点击触发音效（排除音乐按钮，避免冲突）
        document.body.addEventListener('click', (e) => {
            if (e.target.id !== 'musicBtn') {
                initClickSound();
                playClickSound();
            }
        });

        // ========== 粒子背景 + 文字滚动动画：保留原有效果 ==========
        particlesJS('particles-js', {
            particles: {
                number: { value: 70, density: { enable: true, value_area: 800 } },
                color: { value: "#6d5b97" },
                shape: { type: "circle" },
                opacity: { value: 0.3, random: true },
                size: { value: 3, random: true },
                move: {
                    enable: true,
                    speed: 0.5,
                    direction: "none",
                    random: true,
                    out_mode: "out"
                }
            },
            interactivity: {
                detect_on: "canvas",
                events: {
                    onhover: { enable: true, mode: "repulse" },
                    onclick: { enable: true, mode: "push" }
                }
            }
        });

        // 文字滚动显隐动画
        const poemLines = document.querySelectorAll('.poem-line');
        function checkLineVisibility() {
            poemLines.forEach(line => {
                const rect = line.getBoundingClientRect();
                if (rect.bottom > 0 && rect.top < window.innerHeight) {
                    line.classList.add('visible');
                }
            });
        }
        window.addEventListener('load', () => setTimeout(checkLineVisibility, 100));
        window.addEventListener('scroll', checkLineVisibility);
        window.addEventListener('resize', checkLineVisibility);
    </script>
